fn fib(n: u32) -> u32 {
    if n < 2 {
        // Базовый случай: если n равно 0 или 1, возвращаем n.
        return n;
    } else {
        // Рекурсивный случай: возвращаем сумму двух предыдущих чисел Фибоначчи.
        return fib(n - 1) + fib(n - 2);
    }
}

fn main() {
    let n = 20;
    println!("fib({n}) = {}", fib(n));
}
